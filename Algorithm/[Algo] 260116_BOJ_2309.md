# 오늘의 알고리즘
## BAEKJOON 2309번
```python
dwarfs = []

for _ in range(9):
    height = int(input())
    dwarfs.append(height)
    
total_sum = sum(dwarfs)
# dwarfs = [int(input()) for _ in range(9)]

fake_1 = 0
fake_2 = 0

for i in range(9):
    for j in range(i+1, 9):
        if total_sum - (dwarfs[i] + dwarfs[j]) == 100:
            fake_1 = dwarfs[i]
            fake_2 = dwarfs[j]
            break
dwarfs.remove(fake_1)
dwarfs.remove(fake_2)

dwarfs.sort()

for d in dwarfs:
    print(d)
```
